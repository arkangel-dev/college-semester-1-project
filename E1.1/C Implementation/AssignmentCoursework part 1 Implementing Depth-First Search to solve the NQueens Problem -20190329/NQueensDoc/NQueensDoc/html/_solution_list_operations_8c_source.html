36 int GetIndexOfWorkingCandidateInThisList(candidateList thislist)
37 {
38  extern candidateSolution workingCandidate;
39  char wcvalues[N];
40  int place=0, which=0, mismatches=N;
41 
42  //copy peg values from working candidate into conveniently named array becuase it is faster this way
43  for(place=0;place<N;place++)
44  wcvalues[place] = workingCandidate.variableValues[place];
45 
46  //start at the end of the list
47  which = thislist.indexOfLastEntryAdded;
48  //and loop through until we have found a match or reached the beginning
49  while (which >=0)
50  {
51  // start looking peg by peg, stopping as soon as there is a mismatch
52  for (place=0,mismatches = 0; (place < N)&& (mismatches ==0);place++)
53  if (thislist.listEntries[which].variableValues[place] != wcvalues[place])
54  mismatches++;
55  if (mismatches>0) //no mismatches found so move to next one in the list
56  which --;
57  else
58  break;
59  }
60  //end of which loop, now just need to look and see if we found it
61  if (which < 0)
62  which = NOTFOUND;
63  return which;
64 }

233 void RemoveFromListParam1_CandidateSolutionAtIndexParam2( candidateList *listToEdit, int solutionIndex)
234 {
235 
236  int numberOfEntriesInThisList; //declare int to hold number of solutions currently in the list
237  int copyingfrom, copyingto;// and two more to use when we shuffle things along
238  int row=0;
239 
240  //next we'll start off with a series of error checks before we do anything
241  //each of these is serous enough to cause us to quit the programme
242 
243  //first of all, have we been given the address of a valid candidate list
244  if( listToEdit == (candidateList *)(NULL))
245  PrintThisMessageAndExit("first parameter passed was not a pointer to a valid list");
246 
247  // then of all check that the index we are given is valid
248  else if ( (solutionIndex >= SIZEOFSOLUTIONLIST) || (solutionIndex <0))
249  PrintThisMessageAndExit("oops - tried to remove a solution that lies beyond the bounds of our list");
250 
251  // if we've got here we know it is safe to try and access the lists
252  else
253  {
254  // last couple of tests
255  //find out how many entries are in the list
256  numberOfEntriesInThisList = listToEdit->indexOfLastEntryAdded;
257  // check that the list is not already empty
258  if (numberOfEntriesInThisList<0)
259  PrintThisMessageAndExit("oops - tried to remove an entry from an empty list");
260  //check the index isn't too high
261  if ( solutionIndex > numberOfEntriesInThisList)
262  PrintThisMessageAndExit("oops - tried to remove a solution that does not exist in our list");
263 
264 
265 
266  // ok, then we can proceed
267  printf("removing solution at index %d \n", solutionIndex);
268  //if the one to remove is the last on the list, its all very simple
269  if (solutionIndex == numberOfEntriesInThisList)
270  {
271  //empty the contents of the place in the lsit occupied by the last solution added
272  listToEdit->listEntries[solutionIndex].numberOfDefinedValues = 0;
273  listToEdit->listEntries[solutionIndex].score = 0;
274  for ( row=0;row < N;row++)
275  listToEdit->listEntries[solutionIndex].variableValues[row] = UNUSED;
276  //and reduce the count of number of things in the list
277  listToEdit->indexOfLastEntryAdded --;
278  }//end of code that just deleted the last entry
279 
280 
281  else //otherwise we need to shuffle all the rest along,
282  {
283  //start with the one to be deleted, and overwrite the contents with the solution to the right
284  //then move right and repeat
285  for(copyingto = solutionIndex; copyingto < listToEdit->indexOfLastEntryAdded;copyingto++)
286  {
287  copyingfrom = copyingto + 1;//overwrite with the contents of the solution to the right in the list
288 
289  listToEdit->listEntries[copyingto].numberOfDefinedValues =
290  listToEdit->listEntries[copyingfrom].numberOfDefinedValues;
291 
292  listToEdit->listEntries[copyingto].score =
293  listToEdit->listEntries[copyingfrom].score;
294  for (row=0;row < N;row++)
295  listToEdit->listEntries[copyingto].variableValues[row] =
296  listToEdit->listEntries[copyingfrom].variableValues[row];
297 
298  }//end of copying loop
299  }//end if code that deals with shuffling situation
300 
301  }//end of code for valid call
302 }//end of generic removefunction
