21 void CalculateNumberOfVulnerableQueensForWorkingCandidate()
22 {
23  int howManyAreVulnerable = 0; //counter to store the number of vulnerable queens in this solution
24  int attackingQueen; //loop variable - we will consider the queens one by one
25  int defendingQueen; // the queens being attacked
26  int row ; //loop variable
27 
28  extern candidateSolution workingCandidate;
29 
30 
31  //lets make a copy of the relevant bits of the solution we are concerned with
32  // we don;t really need ot do this but it is faster thn repeatedly referring back via a pointer
33  // and avoids the risk of accidntally changing one of the values
34  int columnPositions[N];
35  int numberToConsider = workingCandidate.numberOfDefinedValues;
36  for ( row=0;row < numberToConsider;row++)
37  columnPositions[row] = workingCandidate.variableValues[row];
38 
39 
40  //start by considering the queen in row 0 against all the other defined queens
41  // remember that not all queens may be defined, and that if , say 5 are,
42  // then the last comparison we want to do is whether 4th can take the 5th, or we'll be double counting
43  for (attackingQueen=0; attackingQueen < numberToConsider -1; attackingQueen++)
44  {
45  // so we need to consider the next queen, up to the number defined in this solution
46  for(defendingQueen = attackingQueen+1; defendingQueen < numberToConsider; defendingQueen++)
47  {
48  //are the queens in the same column?
49  if( columnPositions[attackingQueen] == columnPositions[defendingQueen])
50  howManyAreVulnerable ++;
51  // are they on a diagonal running from top left to bottom right?
52  if (( attackingQueen - defendingQueen) == (columnPositions[attackingQueen] - columnPositions[defendingQueen]))
53  howManyAreVulnerable ++;
54  // are they on a diagonal running from top right to bottom left?
55  if (( attackingQueen - defendingQueen ) == ( columnPositions[defendingQueen] - columnPositions[attackingQueen]))
56  howManyAreVulnerable ++;
57  }
58  }
59 
60  //we'e looped through all the queens so now we can store the value
61  workingCandidate.score = howManyAreVulnerable;
62 
63 
64  //finally increment the numberof soltions that have bene tested
65  numberOfCallsMade++;
66 }
